two_sum.py
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        # Store values we've seen: val -> index
        prevMap = {}
        
        for i, n in enumerate(nums):
            diff = target - n
            if diff in prevMap:
                return [prevMap[diff], i]
            
            # Add to map and continue
            prevMap[n] = i
            
        return []
